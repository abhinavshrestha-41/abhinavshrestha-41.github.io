	<script type="module">
		import * as THREE from "../../libs/three.js/build/three.module.js";
	
		// Initialize Potree viewer
		window.viewer = new Potree.Viewer(document.getElementById("potree_render_area"));
	
		viewer.setEDLEnabled(true);
		viewer.setFOV(60);
		viewer.setPointBudget(2_000_000);
		
		// Load settings from URL if needed
		viewer.loadSettingsFromURL();
		viewer.setDescription("");
	
		viewer.loadGUI(() => {
			viewer.setLanguage("en");
			$("#menu_appearance").next().show();
			$("#menu_tools").next().show();
			$("#menu_clipping").next().show();
			viewer.toggleSidebar();
		});
	
		// Load point cloud
		Potree.loadPointCloud("./pointclouds/MSThesisRFClass/metadata.json", "MSThesisRFClass", e => {
			let scene = viewer.scene;
			let pointcloud = e.pointcloud;
	
			let material = pointcloud.material;
			material.size = 1;
			material.pointSizeType = Potree.PointSizeType.ADAPTIVE;
			material.shape = Potree.PointShape.CIRCLE;
			material.activeAttributeName = "classification";
	
			// Add point cloud to scene
			scene.addPointCloud(pointcloud);
	
			// Apply custom classification scheme
			setCustomScheme();
			
			// Set initial camera position (your current view position, set from browser console with `console.log(viewer.scene.view.position);`)
			viewer.scene.view.position.set(289033.95624412294, 5204491.008589532, 51.73155726597741);

			// Set the look-at target (pivot point where the camera is looking, set from browser console with `console.log(viewer.scene.view.getPivot());`)
			viewer.scene.view.lookAt(289032.5238838257, 5204567.8504722975, 15.745850918022235);

			// Set background and fit view
			viewer.setBackground("white", "white");
			// viewer.fitToScreen();
			
		
		});
	
		// Define a function to set the custom classification scheme
		function setCustomScheme() {
			const customClassification = {
				1: { visible: true, name: "Class 1", color: [0.13, 0.55, 0.13, 1] },  // Forest Green
				2: { visible: true, name: "Class 2", color: [0.7, 0.7, 0.7, 1] },  // Brighter Gray
				3: { visible: true, name: "Class 3", color: [0.65, 0.16, 0.16, 1] },  // Brownish-Red (Insect Damage)
				4: { visible: true, name: "Class 4", color: [0, 0, 0, 1] },  // Black (Shadow)
				DEFAULT: { visible: true, name: "Default", color: [1, 1, 1, 1] }  // White
			};
	
			viewer.setClassifications(customClassification);
		
			console.log("Applied Custom Classification:", viewer.classifications);
		}
	
	</script>
